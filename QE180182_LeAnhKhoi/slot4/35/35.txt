import java.util.Scanner;

public class Main {
public static void main (String[] args) {
Scanner sc = new Scanner(System.in);
int n = sc.nextInt();
int[] a = new int[n];
for(int i = 0; i < n; i++) {
a[i] = sc.nextInt();
}
boolean b = true;
for(int i = 0; i < n; i+=2) {
if(a[i] % 2 == 1) {
b = false;
System.out.print(a[i] + " ");
}
}
if(b) {
System.out.print("NULL");
}
}
}
